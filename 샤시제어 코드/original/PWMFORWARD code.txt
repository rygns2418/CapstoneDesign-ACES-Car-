
#include <stdio.h>
#include <stdlib.h>
#include <pigpio.h>

// BACK 동작 조건때문에 FORWARD도 동일 포맷으로 코딩
int main()
{


	if(gpioInitialise()<0) return -1;

gpioSetMode(17,PI_OUTPUT);
gpioSetPWMfrequency(17,40);
gpioSetPWMrange(17,400);

gpioPWM(17, 24);
time_sleep(0.05); 
gpioPWM(17, 25); 
time_sleep(0.05);
gpioPWM(17, 0);
time_sleep(0.05);

gpioPWM(17, 24);
time_sleep(0.05);
gpioPWM(17, 25);
time_sleep(0.05);

int A = gpioGetPWMrange(17);
int B = gpioGetPWMdutycycle(17);
int C = gpioGetPWMfrequency(17);
printf("rage : %d\n",A);
printf("duty cycle : %d\n",B);
printf("frequency : %d\n",C);

while(1)
{}
return 0;

}
